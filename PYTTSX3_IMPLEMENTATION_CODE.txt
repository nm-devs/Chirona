"""
Text-to-Speech Module using pyttsx3

Provides a thread-safe, non-blocking text-to-speech system for the application.
Allows offline speech synthesis with configurable speech rate and volume.
"""

import pyttsx3
import queue
import threading
import logging
from typing import Optional

# Configure logging
logger = logging.getLogger(__name__)


class TextToSpeech:
    """
    A thread-safe wrapper around pyttsx3 for non-blocking text-to-speech.
    
    Features:
    - Non-blocking speech via background worker thread
    - Configurable speech rate (default 150 wpm) and volume (0-1)
    - Graceful fallback if TTS engine is unavailable
    - Thread-safe queue for managing speech requests
    - Clean shutdown and resource management
    """
    
    def __init__(self, rate: int = 150, volume: float = 0.9):
        """
        Initialize the TextToSpeech engine.
        
        Args:
            rate (int): Words per minute for speech (default 150)
            volume (float): Volume level 0-1 (default 0.9)
        """
        self.rate = rate
        self.volume = volume
        self.engine = None
        self.is_available = False
        self.speech_queue = queue.Queue(maxsize=50)  # Prevent unlimited growth
        self.should_stop = False
        self.worker_thread = None
        self.lock = threading.Lock()
        
        self._initialize_engine()
        
    def _initialize_engine(self):
        """Initialize the pyttsx3 engine with error handling."""
        try:
            self.engine = pyttsx3.init()
            self.engine.setProperty('rate', self.rate)
            self.engine.setProperty('volume', self.volume)
            
            # Try to set a preferred voice (female if available)
            voices = self.engine.getProperty('voices')
            if len(voices) > 1:
                # Prefer female voice (usually index 1)
                self.engine.setProperty('voice', voices[1].id)
            elif len(voices) > 0:
                self.engine.setProperty('voice', voices[0].id)
            
            self.is_available = True
            logger.info(f"TTS Engine initialized successfully. "
                       f"Rate: {self.rate} wpm, Volume: {self.volume}")
            
            # Start worker thread
            self.should_stop = False
            self.worker_thread = threading.Thread(
                target=self._process_speech_queue,
                daemon=True
            )
            self.worker_thread.start()
            logger.info("TTS worker thread started")
            
        except Exception as e:
            self.is_available = False
            logger.warning(f"TTS engine initialization failed: {e}. "
                          f"Speech will be disabled.")
    
    def speak(self, text: str) -> bool:
        """
        Queue text for speech synthesis (non-blocking).
        
        Args:
            text (str): Text to speak
            
        Returns:
            bool: True if queued successfully, False if engine unavailable
        """
        if not text or not self.is_available:
            return False
        
        try:
            # Don't block on queue.put() to maintain responsiveness
            self.speech_queue.put_nowait(text)
            logger.debug(f"Queued for speech: '{text}'")
            return True
        except queue.Full:
            logger.warning("Speech queue is full. Dropping speech request.")
            return False
    
    def _process_speech_queue(self):
        """Background worker thread that processes queued speech requests."""
        logger.debug("Speech worker thread started")
        
        while not self.should_stop:
            try:
                # Wait for a speech request with timeout to check should_stop flag
                text = self.speech_queue.get(timeout=0.5)
                
                if text and self.engine:
                    try:
                        with self.lock:
                            self.engine.say(text)
                            self.engine.runAndWait()
                        logger.debug(f"Speech completed: '{text}'")
                    except Exception as e:
                        logger.error(f"Error during speech synthesis: {e}")
                
                self.speech_queue.task_done()
                
            except queue.Empty:
                # Timeout - continue checking should_stop flag
                continue
            except Exception as e:
                logger.error(f"Error in speech worker thread: {e}")
        
        logger.debug("Speech worker thread stopped")
    
    def set_speech_rate(self, rate: int):
        """
        Set the speech rate in words per minute.
        
        Args:
            rate (int): Words per minute (typically 50-300)
        """
        self.rate = max(50, min(300, rate))  # Clamp to reasonable range
        if self.engine and self.is_available:
            with self.lock:
                self.engine.setProperty('rate', self.rate)
            logger.info(f"Speech rate set to {self.rate} wpm")
    
    def set_volume(self, volume: float):
        """
        Set the speech volume level.
        
        Args:
            volume (float): Volume level 0-1
        """
        self.volume = max(0.0, min(1.0, volume))  # Clamp to 0-1
        if self.engine and self.is_available:
            with self.lock:
                self.engine.setProperty('volume', self.volume)
            logger.info(f"Volume set to {self.volume}")
    
    def get_available_voices(self):
        """
        Get list of available voices on the system.
        
        Returns:
            list: List of voice objects with id and name properties
        """
        if self.engine and self.is_available:
            try:
                return self.engine.getProperty('voices')
            except Exception as e:
                logger.error(f"Error getting voices: {e}")
        return []
    
    def set_voice(self, voice_id: str):
        """
        Set the voice to use for speech synthesis.
        
        Args:
            voice_id (str): ID of the voice to use
        """
        if self.engine and self.is_available:
            try:
                with self.lock:
                    self.engine.setProperty('voice', voice_id)
                logger.info(f"Voice set to {voice_id}")
            except Exception as e:
                logger.error(f"Error setting voice: {e}")
    
    def clear_queue(self):
        """Clear all pending speech requests from the queue."""
        while not self.speech_queue.empty():
            try:
                self.speech_queue.get_nowait()
            except queue.Empty:
                break
        logger.info("Speech queue cleared")
    
    def shutdown(self):
        """Cleanly shutdown the TTS engine and worker thread."""
        logger.info("Shutting down TTS engine...")
        
        self.should_stop = True
        
        # Wait for worker thread to finish
        if self.worker_thread and self.worker_thread.is_alive():
            self.worker_thread.join(timeout=2.0)
            if self.worker_thread.is_alive():
                logger.warning("Worker thread did not stop gracefully")
        
        # Cleanup engine
        if self.engine:
            try:
                self.engine.stop()
            except Exception as e:
                logger.warning(f"Error stopping engine: {e}")
            self.engine = None
        
        self.is_available = False
        logger.info("TTS engine shutdown complete")
    
    def __del__(self):
        """Ensure cleanup on object destruction."""
        self.shutdown()

"""
Controller for processing visual feedback in sign language translation mode.

Handles the rendering of hand tracking points and skeletons on the camera feed
when the user is spelling out signs. Includes text-to-speech functionality for
spoken feedback of recognized signs.
"""

import cv2
import logging
from utils.text_to_speech import TextToSpeech
from utils.drawing_utils import draw_hand_points, draw_hand_skeleton
from config import TTS_ENABLED, TTS_SPEAK_LETTERS, TTS_SPEAK_WORDS

logger = logging.getLogger(__name__)


class SignLanguageController:
    """
    Controller for sign language recognition mode.
    
    Manages visual feedback and text-to-speech for recognized signs and words.
    """
    
    def __init__(self, tts: TextToSpeech = None):
        """
        Initialize the sign language controller.
        
        Args:
            tts (TextToSpeech): Optional TextToSpeech instance. 
                               If None, creates a new one.
        """
        if tts:
            self.tts = tts
        else:
            self.tts = TextToSpeech() if TTS_ENABLED else None
        
        self.last_spoken_sign = None
        self.last_spoken_word = None
        logger.info("SignLanguageController initialized")

    def process_frame(self, frame, hand, detector):
        """
        Process one frame in sign-language mode.
        
        Args:
            frame: OpenCV image frame
            hand: Hand data dictionary with positions and landmarks
            detector: HandDetector instance
        """
        positions = hand["positions"]
        hand_landmarks = hand["landmarks"]

        # Draw visuals
        draw_hand_points(frame, positions)
        draw_hand_skeleton(frame, hand_landmarks, detector.mp_hands, detector.mp_drawing)

        # Placeholder text
        cv2.putText(
            frame,
            "Sign Language Mode - Ready",
            (10, 60),
            cv2.FONT_HERSHEY_SIMPLEX,
            0.7,
            (0, 255, 255),
            2,
        )

    def speak_sign(self, sign: str):
        """
        Speak a recognized sign using TTS.
        
        Args:
            sign (str): The sign/letter to speak
        """
        if not sign or not self.tts or not TTS_ENABLED:
            return
        
        # Avoid repeated speech of the same sign
        if sign == self.last_spoken_sign and TTS_SPEAK_LETTERS:
            return
        
        if TTS_SPEAK_LETTERS:
            self.tts.speak(sign)
            self.last_spoken_sign = sign
            logger.debug(f"Speaking sign: {sign}")

    def speak_word(self, word: str):
        """
        Speak a confirmed word using TTS.
        
        Args:
            word (str): The word to speak
        """
        if not word or not self.tts or not TTS_ENABLED:
            return
        
        # Avoid repeated speech of the same word
        if word == self.last_spoken_word and TTS_SPEAK_WORDS:
            return
        
        if TTS_SPEAK_WORDS:
            self.tts.speak(word)
            self.last_spoken_word = word
            logger.info(f"Speaking word: {word}")

    def speak_sentence(self, sentence: str):
        """
        Speak a complete sentence using TTS.
        Triggered by thumbs-up gesture or manual trigger.
        
        Args:
            sentence (str): The sentence to speak
        """
        if not sentence or not self.tts or not TTS_ENABLED:
            return
        
        self.tts.speak(sentence)
        logger.info(f"Speaking sentence: {sentence}")

    def handle_thumbs_up_gesture(self, current_sentence: str):
        """
        Handle the thumbs-up gesture to trigger speech.
        
        Args:
            current_sentence (str): The sentence to speak
        """
        if current_sentence:
            self.speak_sentence(current_sentence)
            logger.info("Thumbs-up gesture: speaking sentence")

    def clear_speech_memory(self):
        """Clear the memory of spoken items to allow re-speaking."""
        self.last_spoken_sign = None
        self.last_spoken_word = None

    def shutdown(self):
        """Shutdown the TTS engine."""
        if self.tts:
            self.tts.shutdown()
            logger.info("SignLanguageController shutdown")

    def _handle_keypress(self):

        key = cv2.waitKey(1) & 0xFF
        
        # Toggle control modes

            self.mode = "sign_language" if self.mode == "mouse" else "mouse"
            self.sentence_builder.clear()
            
        # Manually add space with spacebar when in sign language mode
        if key == ord(' ') and self.mode == "sign_language":
            self.sentence_builder.add_space()
        
        # Trigger speech with 's' key (in addition to thumbs-up gesture)
        if key == ord('s') and self.mode == "sign_language":
            sentence = self.sentence_builder.get_current_sentence()
            if sentence and self.controllers["sign_language"].tts:
                self.controllers["sign_language"].speak_sentence(sentence)
                logging.info(f"Speech triggered: {sentence}")
            
        if key == ord('h'):
            # toggle between 1 and 2 hand modes
            self.max_hands_mode = 2 if self.max_hands_mode == 1 else 1
            self.detector.hands.max_num_hands = self.max_hands_mode
            print(f'Hand detection mode: {self.max_hands_mode} hand(s)')
            
        # Exit on Escape key or window close
        if key == 27 or cv2.getWindowProperty(WINDOW_TITLE, cv2.WND_PROP_VISIBLE) < 1:
            return False
